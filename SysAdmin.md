  ````bash
  cd chkrootkit-0.54/

  ./chkrootkit -r /mnt/                         # specifies new root directory
  ./chkrootkit -p /cdrom/bin:/floppy/mybin      # adds binary paths
  ./chkrootkit -x | more
  ./chkrootkit -x | egrep '^/'                  # pathnames inside system commands
  ````
